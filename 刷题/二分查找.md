### 1. 在旋转的升序数组中查找目标
##### 二分法，数组。
```go
func search(nums []int, target int) int {
    l, r := 0, len(nums)-1
    var m int
    for l <= r {
        // 二分法
        m = (l+r)/2
        if nums[m] == target {
            return m
        }
        if nums[l] <= nums[m] {
            // 左边单调增
            if target >= nums[l] && target < nums[m] {
                // 在左边范围
                r = m - 1
            } else {
                // 在右边范围
                l = m + 1
            }
        } else {
            // 右边单调增
            if target > nums[m] && target <= nums[r] {
                // 在右边范围
                l = m + 1
            } else {
                // 在左边范围
                r = m - 1
            }
        }
    }
    // 不存在，返回-1
    return -1
}
//使用二分法，每次都去寻找单调递增的一边；
//如果对应的数在单调递增范围内，则接着缩小范围，直至找到；
//如果不在，则继续步骤1，接着步骤2
```
### 2. 在升序数组中查找目标出现的第一个和最后一个位置
##### [力扣34](https://leetcode-cn.com/problems/find-first-and-last-position-of-element-in-sorted-array/)
##### 二分法，数组
```go
func searchRange(nums []int, target int) []int {
    if len(nums) == 0 {
        return []int{-1,-1}
    }
    l, r := 0, len(nums)-1
    var m int
    for l <= r {
        // 二分法
        m = (l+r)/2
        if target == nums[m] {
            l = m 
            r = m 
            for l-1 >= 0 && target == nums[l-1]{
                l --
            }
            for r+1 <= len(nums)-1 && target == nums[r+1] {
                r ++
            }
            return []int{l,r}
        } else if target > nums[m] {
            // 在右边
            l = m + 1
        } else {
            // 在左边
            r = m - 1
        }
    }
    return []int{-1,-1}
}
```
